import React, { Component } from "react";
import "./SidebarContainer.css";
import PropTypes from "prop-types";

import {
    Tab, Box,
} from "@mui/material";
import {
    TabContext, TabPanel, TabList,
} from "@mui/lab";

/**
 * Component for SidebarContainer.
 * @module SidebarContainer
 * @summary Contains full functions page
 * @extends React.Component
 */
class SidebarContainer extends Component {
    /**
     * @param {Object} props properities given by parent component
     */
    constructor(props) {
        super(props);
        this.state = {
            curActivPanelId: 0,
        };
        this.handleCurActivPanelIdChange = this.handleCurActivPanelIdChange.bind(this);
    }

    componentDidCatch() {
    }

    /**
     * @param {any} event
     * @param {number} newValue newValue for curActivPanelId.
     * If newValue>length of pages OR newValue<0, throw an error.
     */
    handleCurActivPanelIdChange(event, newValue) {
        const { panels } = this.props;
        if (!(panels.length <= 0 && newValue < panels.length)) {
            throw Error("Invalid value for current activated panel Id in SidebarContainer.");
        } else {
            this.setState({
                curActivPanelId: newValue,
            });
        }
    }

    /**
     * Render the panels
     * @returns {React.ElementType}
     */
    render() {
        const { curActivPanelId } = this.state;
        return (
            <TabContext value={curActivPanelId}>
                <Box sx={{ width: "100%" }}>
                    <TabList
                        value={curActivPanelId}
                        onChange={this.handleCurActivPanelIdChange}
                        textColor="secondary"
                        indicatorColor="secondary"
                        aria-label="secondary tabs example"
                    >
                        <Tab value={0} label="Item One" />
                        <Tab value={1} label="Item Two" />
                        <Tab value={2} label="Item Three" />
                    </TabList>
                    <TabPanel value={0}>123</TabPanel>
                    <TabPanel value={1}>456</TabPanel>
                    <TabPanel value={2}>789</TabPanel>
                </Box>
            </TabContext>
        );
    }
}

SidebarContainer.propTypes = {
    panels: PropTypes.arrayOf(PropTypes.shape({
        id: PropTypes.string,
        panel: PropTypes.element,
    })),
};
SidebarContainer.defaultProps = {
    panels: [],
};
export default SidebarContainer;
